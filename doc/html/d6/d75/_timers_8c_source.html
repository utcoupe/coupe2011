00026 //Timer1_Init() sets up Timer1 to count up to the maximum 16-bit value, 0xFFFF,
00027 //and interrupt the CPU. ISR processing is enabled for Timer 1.
00028 void Timer1_Init(void)
00029 {
00030         T1CON = 0x0000;         //Timer1 set up to count on instruction cycle
00031                                 //edge with 1:1 prescaler applied initially.
00032         PR1 = 0xFFFF;           //Period Register, PR1, set to maximum count
00033         IFS0bits.T1IF = 0;      //Clear the Timer1 Interrupt Flag
00034         IEC0bits.T1IE = 1;      //Enable Timer1 Interrupt Service Routine
00035         T1CONbits.TON=1;        //Start Timer 1
00036 }

00038 //Timer2_Init() sets up Timer2 and Timer3 to count up to a 32-bit value,
00039 //0x003FFFFF with a prescaler of 1:1. 32-bit Timer ISR is disabled.
00040 void Timer2_Init(void)
00041 {
00042         T2CON = 0x0000;         //32-bit Timer3:Timer2 pair set up
00043         T2CONbits.T32 = 1;      //to increment every instruction cycle
00044         PR3 = 0x003F;
00045         PR2 = 0xFFFF;           //Period Register, PR3:PR2, set to 0x003FFFFF
00046         IFS0bits.T3IF = 0;      //Clear the Timer3 Interrupt Flag
00047         IEC0bits.T3IE = 0;      //Disable Timer3 Interrup Service Routine
00048         T2CONbits.TON=1;        //Start 32-bit timer, setting Timer 2 ON bit
00049 }
