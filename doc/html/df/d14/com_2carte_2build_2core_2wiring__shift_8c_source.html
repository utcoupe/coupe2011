00027 uint8_t shiftIn(uint8_t dataPin, uint8_t clockPin, uint8_t bitOrder) {
00028         uint8_t value = 0;
00029         uint8_t i;
00030 
00031         for (i = 0; i < 8; ++i) {
00032                 digitalWrite(clockPin, HIGH);
00033                 if (bitOrder == LSBFIRST)
00034                         value |= digitalRead(dataPin) << i;
00035                 else
00036                         value |= digitalRead(dataPin) << (7 - i);
00037                 digitalWrite(clockPin, LOW);
00038         }
00039         return value;
00040 }
