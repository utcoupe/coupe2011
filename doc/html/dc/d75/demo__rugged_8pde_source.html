00131 void loop()
00132 {
00133   if (Serial.available()) {
00134     switch (Serial.read()) {
00135       case 'f': case 'F': DIR = 1; break;
00136       case 'b': case 'B': DIR = -1; break;
00137       case 'o': case 'O':
00138                           if (PWM <= 250) PWM += 5;
00139                           Serial.print("PWM="); Serial.println(PWM);
00140                           analogWrite(EN1_PIN, PWM); analogWrite(EN2_PIN, PWM);
00141                           break;
00142       case 'm': case 'M':
00143                           if (PWM >= 5) PWM -= 5;
00144                           Serial.print("PWM="); Serial.println(PWM);
00145                           analogWrite(EN1_PIN, PWM); analogWrite(EN2_PIN, PWM);
00146                           break;
00147       case 'a': case 'A':
00148                           RPM++;
00149                           Serial.print("RPM="); Serial.println(RPM);
00150                           stepper.setSpeed(RPM);
00151                           break;
00152       case 'd': case 'D':
00153                           if (RPM > 1) RPM--;
00154                           Serial.print("RPM="); Serial.println(RPM);
00155                           stepper.setSpeed(RPM);
00156                           break;
00157       case 'h': case 'H':
00158                           printHelp();
00159                           break;
00160       default:
00161                           Serial.println("Unknown command");
00162                           break;
00163     }
00164   }
00165 
00166   // This is a busy-wait loop until the inter-step time passes
00167   stepper.step(DIR);
00168 
00169   // Mirror phase outputs in LED's
00170   digitalWrite(LED1_PIN, digitalRead(DIR1_PIN));
00171   digitalWrite(LED3_PIN, digitalRead(DIR2_PIN));
00172 
00173   // Drive fault output LED's as mirror of fault inputs
00174   digitalWrite(LED0_PIN, !digitalRead(FAULT1_PIN));
00175   digitalWrite(LED2_PIN, !digitalRead(FAULT2_PIN));
00176 }
