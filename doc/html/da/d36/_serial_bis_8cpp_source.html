00063 inline void store_char(unsigned char c, ring_buffer *rx_buffer)
00064 {
00065   int i = (unsigned int)(rx_buffer->head + 1) % RX_BUFFER_SIZE;
00066 
00067   // if we should be storing the received character into the location
00068   // just before the tail (meaning that the head would advance to the
00069   // current location of the tail), we're about to overflow the buffer
00070   // and so we don't write the character or advance the head.
00071   if (i != rx_buffer->tail) {
00072     rx_buffer->buffer[rx_buffer->head] = c;
00073     rx_buffer->head = i;
00074   }
00075 }

00179 void HardwareSerialBis::begin(long baud)
00180 {
00181   uint16_t baud_setting;
00182   bool use_u2x = true;
00183 
00184 #if F_CPU == 16000000UL
00185   // hardcoded exception for compatibility with the bootloader shipped
00186   // with the Duemilanove and previous boards and the firmware on the 8U2
00187   // on the Uno and Mega 2560.
00188   if (baud == 57600) {
00189     use_u2x = false;
00190   }
00191 #endif
00192   
00193   if (use_u2x) {
00194     *_ucsra = 1 << _u2x;
00195     baud_setting = (F_CPU / 4 / baud - 1) / 2;
00196   } else {
00197     *_ucsra = 0;
00198     baud_setting = (F_CPU / 8 / baud - 1) / 2;
00199   }
00200 
00201   // assign the baud_setting, a.k.a. ubbr (USART Baud Rate Register)
00202   *_ubrrh = baud_setting >> 8;
00203   *_ubrrl = baud_setting;
00204 
00205   sbi(*_ucsrb, _rxen);
00206   sbi(*_ucsrb, _txen);
00207   sbi(*_ucsrb, _rxcie);
00208 }

00231 int HardwareSerialBis::read(void)
00232 {
00233   // if the head isn't ahead of the tail, we don't have any characters
00234   if (_rx_buffer->head == _rx_buffer->tail) {
00235     return -1;
00236   } else {
00237     unsigned char c = _rx_buffer->buffer[_rx_buffer->tail];
00238     _rx_buffer->tail = (unsigned int)(_rx_buffer->tail + 1) % RX_BUFFER_SIZE;
00239     return c;
00240   }
00241 }

00243 void HardwareSerialBis::flush()
00244 {
00245   // don't reverse this or there may be problems if the RX interrupt
00246   // occurs after reading the value of rx_buffer_head but before writing
00247   // the value to rx_buffer_tail; the previous value of rx_buffer_head
00248   // may be written to rx_buffer_tail, making it appear as if the buffer
00249   // don't reverse this or there may be problems if the RX interrupt
00250   // occurs after reading the value of rx_buffer_head but before writing
00251   // the value to rx_buffer_tail; the previous value of rx_buffer_head
00252   // may be written to rx_buffer_tail, making it appear as if the buffer
00253   // were full, not empty.
00254   _rx_buffer->head = _rx_buffer->tail;
00255 }
