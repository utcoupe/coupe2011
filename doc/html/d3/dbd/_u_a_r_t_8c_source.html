00055 //UART_Init() sets up the UART for a 8-bit data, No Parity, 1 Stop bit
00056 //at 9600 baud with transmitter interrupts enabled
00057 void UART_Init (void)
00058 {
00059         U1MODE = 0x0000;        //Clear UART2 registers
00060         U1STA = 0x0000;
00061         U1MODEbits.ALTIO = 1;   //Enable U1ATX and U1ARX instead of
00062                                 //U1TX and U1RX pins
00063 
00064         U1MODEbits.UARTEN = 1;  //Enable UART1 module
00065         U1BRG = BRGVAL;         //Load UART1 Baud Rate Generator
00066 
00067         IFS0bits.U1RXIF = 0;    //Clear UART1 Receiver Interrupt Flag
00068         IFS0bits.U1TXIF = 0;    //Clear UART1 Transmitter Interrupt Flag
00069         IEC0bits.U1RXIE = 0;    //Disable UART1 Receiver ISR
00070         IEC0bits.U1TXIE = 1;    //Enable UART1 Transmitter ISR
00071         U1STAbits.UTXISEL = 1;  //Setup UART1 transmitter to interrupt
00072                                 //when a character is transferred to the
00073                                 //Transmit Shift register and as result,
00074                                 //the transmit buffer becomes empty.
00075 
00076         U1STAbits.UTXEN = 1;    //Enable UART1 transmitter
00077         UARTCharPtr = &DisplayData[0]; //Initialize UARTCharPtr to point
00078                                 //to the first character in the Display buffer
00079 }

00081 //WriteUART_to_RS232() triggers interrupt-driven UART communication by writing
00082 //the first character in the Display buffer to the UART Transmit register
00083 void WriteUART_to_RS232(void)
00084 {
00085         if ((UARTCharPtr > &DisplayData[0]) &&
00086                 (UARTCharPtr < &DisplayData[38])) return;
00087         else
00088         {
00089                 UARTCharPtr = &DisplayData[0];  //Re-Initialize UART display
00090                                                 //buffer pointer to point to
00091                                                 //the first character
00092                 U1TXREG = *UARTCharPtr++;       //Load the UART transmit
00093                                                 //register with first character
00094         }
00095 }

00098 //_U1TXInterrupt() is the UART1 Interrupt Service Routine.
00099 //The routine must have global scope in order to be an ISR.
00100 //The ISR name is the same name provided for the module in the device linker
00101 //script.
00102 //The UART1 ISR loads the UART2 4-deep FIFO buffers with the next
00103 //4 characters in the Display buffer unless it encounters a null character.
00104 void __attribute__((__interrupt__, __no_auto_psv__)) _U1TXInterrupt(void)
00105 {
00106         int i = 0;
00107         while ((*UARTCharPtr != '\0') && (i < 4))
00108         {
00109                 U1TXREG = *UARTCharPtr++;
00110                 i++;
00111         }
00112         IFS0bits.U1TXIF = 0;    //Clear the UART1 transmitter interrupt flag
00113 }
